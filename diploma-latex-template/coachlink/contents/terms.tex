\newglossaryentry{backend-platform}{
    name={Backend-платформа},
    description={серверная часть программной системы, включающая бизнес-логику, API, хранение данных и межсервисное взаимодействие}
}

\newglossaryentry{microservice}{
    name={Микросервис},
    description={независимый сервис с ограниченной зоной ответственности, взаимодействующий с другими сервисами через API или сообщения}
}

\newglossaryentry{microservice-architecture}{
    name={Микросервисная архитектура},
    description={архитектурный подход, при котором система строится как набор независимых сервисов, развиваемых и разворачиваемых отдельно}
}

\newglossaryentry{event-driven}{
    name={Событийное взаимодействие},
    description={асинхронный способ обмена, при котором сервис публикует событие, а подписчики обрабатывают его независимо}
}

\newglossaryentry{stateless}{
    name={Stateless-сервис},
    description={сервис, не владеющий собственной предметной базой данных и вычисляющий результат по входным данным и данным других сервисов}
}

\newglossaryentry{public-api}{
    name={Публичный API},
    description={внешний программный интерфейс платформы, доступный клиентским приложениям через API Gateway}
}

\newglossaryentry{internal-api}{
    name={Внутренний API},
    description={служебный программный интерфейс межсервисного взаимодействия, недоступный внешним клиентам}
}

\newglossaryentry{api-gateway}{
    name={API Gateway},
    description={единая точка входа в backend-платформу, выполняющая проверку доступа и маршрутизацию запросов к внутренним сервисам}
}

\newglossaryentry{coach-athlete-relation}{
    name={Связь тренер-спортсмен},
    description={подтверждённое отношение между тренером и спортсменом, определяющее права доступа к тренировочным данным}
}

\newglossaryentry{reverse-proxy}{
    name={Reverse proxy},
    description={сервер-посредник, принимающий запросы от клиентов и перенаправляющий их к внутренним сервисам, скрывая их адреса от внешних потребителей}
}

\newglossaryentry{durable-consumer}{
    name={Durable consumer},
    description={подписчик брокера сообщений, сохраняющий позицию чтения между перезапусками и обеспечивающий обработку пропущенных сообщений}
}

\newglossaryentry{middleware}{
    name={Middleware},
    description={промежуточный обработчик в цепочке обработки HTTP-запроса, выполняющий сквозную логику (аутентификация, логирование, CORS) до передачи запроса основному обработчику}
}
